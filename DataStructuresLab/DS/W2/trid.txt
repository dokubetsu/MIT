#include <iostream>

using namespace std;

int main() {
    int size;
    cout<<"Enter the size of the square matrix: ";
    cin>>size;

    int matrix[size][size];

    cout<< "Enter the elements of the tri-diagonal matrix row wise:"<<endl;
    for (int i = 0; i < size; ++i) {
        for (int j = 0; j < size; ++j) {
            if (i == j || i == j + 1 || i == j - 1) {
                cin>>matrix[i][j];
            } else {
                matrix[i][j] = 0;
            }
        }
    }
    int oneDimArray[size * 3 - 2];
    int k = 0;
    for (int i = 0; i < size; ++i) {
        for (int j = i - 1; j <= i + 1; ++j) {
            if (j >= 0 && j < size) {
                oneDimArray[k++] = matrix[i][j];
            }
        }
    }
    cout<< "One-dimensional array representation:"<<endl;
    for (int i = 0; i < (size *3)-2; ++i) {
        cout<<oneDimArray[i]<< " ";
    }
    cout<<endl;

    cout<<"Tri-diagonal matrix representation:"<<endl;
    for (int i = 0; i < size; ++i) {
        for (int j = 0; j < size; ++j) {
            cout << matrix[i][j] << " ";
        }
        cout<<endl;
    }
    int row, col;
    cout<<"Enter the row and column to display element: \n";
    cin>>row>>col;

    if (row >= 0 && row < size && col >= 0 && col < size) {
        if (row == col || row == col + 1 || row == col - 1) {
            int position = row * 3 - (row > col ? row - col : col - row);
            cout<<"Element at position ("<<row<<", "<<col<<"): "<<oneDimArray[position]<<endl;
        } else {
            cout<<"Element at position ("<<row<<", "<<col<<") is 0."<<endl;
        }
    } else {
        cout<<"Invalid position."<<endl;
    }

    return 0;
}
