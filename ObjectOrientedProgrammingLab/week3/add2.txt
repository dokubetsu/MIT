public class SinSeries {
    public static void main(String[] args) {
        double x = 2.0; // Replace with the desired value of x
        int n = 10;     // Number of terms in the series
        double result = 0;

        for (int i = 0; i < n; i++) {
            int exponent = 2 * i + 1;
            double term = Math.pow(-1, i) * Math.pow(x, exponent) / factorial(exponent);
            result += term;
        }

        System.out.println("sin(" + x + ") = " + result);
    }

    public static int factorial(int n) {
        if (n == 0) {
            return 1;
        } else {
            return n * factorial(n - 1);
        }
    }
}


public class SeriesSum {
    public static void main(String[] args) {
        int n = 5; // Number of terms in the series
        double sum = 0;

        for (int i = 1; i <= n; i++) {
            double term = Math.pow(1.0 / i, i);
            sum += term;
        }

        System.out.println("Sum of the series = " + sum);
    }
}
